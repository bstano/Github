def get_length():
    n = 0
    while True:
        user_length = input("What length should the sequence be?\n")
        try:
            length = int(user_length)
            if length >= 2:
                return length
            else:
                print('The length must be a positive integer of 2 or more. Try again.')
                n += 1
        except:
            print('The length should be a positive integer.')
            n += 1
        if n == 3:
            print("You're having trouble, let's set the length to 20")
            return 20
    
    
def generate_fibs(length):
    result = []
    for n in range(length):
        if n == 0:
            result.append(0)
        elif n == 1:
            result.append(1)
        else:
            result.append(result[n-1]+result[n-2])
    return result
    
def display_fibs(fibs):
    for fib in fibs:
        print(fib)
    
def main():
    length = get_length()
    fibs = generate_fibs(length)
    display_fibs(fibs)

main()
